// sắp xếp tăng dần
Insertion_Sort(K, n){
	for (i = 1; i < n; i ++){
		x = K[i];
		j = i – 1;
		while ((K[j] > x) && (j >= 0)){
			//Lùi K[j] lại một vị trí
			K[j + 1] = K[j];
			j --;
		}
		K[j + 1] = x;
	}
}
